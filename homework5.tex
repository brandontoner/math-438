    \begin{proof}[$\frac{1}{n!}\frac{d^n (1+z)^\alpha}{dz^n}={\alpha \choose n}(1+z)^{\alpha-n}$]
        \begin{eqnarray*}
            \textbf{Basis: } \frac{1}{1!}\frac{d (1+z)^\alpha}{dz} &=& \alpha (1+z)^{\alpha-1} \\
            {\alpha \choose 1} (1+z)^{\alpha-1}&=& \alpha (1+z)^{\alpha-1} \\
            \textbf{Assume } \frac{1}{n!}\frac{d^n (1+z)^\alpha}{dz^n}&=&{\alpha \choose n}(1+z)^{\alpha-n} \\
            \frac{d}{dz}\left(\frac{1}{n!}\frac{d^n (1+z)^\alpha}{dz^n}\right) &=& \frac{1}{n!}\frac{d^{n+1} (1+z)^\alpha}{dz^{n+1}} \\
            \frac{d}{dz}\left({\alpha \choose n}(1+z)^{\alpha-n}\right) &=& {\alpha \choose n}(\alpha-n)(1+z)^{\alpha-n-1} \\
            \frac{1}{n!}\frac{d^{n+1} (1+z)^\alpha}{dz^{n+1}} &=& {\alpha \choose n}(\alpha-n)(1+z)^{\alpha-(n+1)} \\
            \frac{1}{(n+1)n!}\frac{d^{n+1} (1+z)^\alpha}{dz^{n+1}} &=& {\alpha \choose n}\frac{\alpha-n}{n+1}(1+z)^{\alpha-(n+1)} \\
            \frac{1}{(n+1)!}\frac{d^{n+1} (1+z)^\alpha}{dz^{n+1}} &=& {\alpha \choose n+1}(1+z)^{\alpha-(n+1)}
        \end{eqnarray*}
    \end{proof}
    \begin{proof}[$\sum\limits_{n=0}^{\infty}{{\alpha \choose n} z^n} = (1+z)^\alpha$]
       The Taylor series for a function, $f(z)$ is:
        \begin{eqnarray*}
            f(z) &=& f(z_0) + \sum\limits_{n=1}^{\infty}{\left[\left.\frac{1}{n!}\frac{d^nf(z)}{dz^n}\right|_{z=z_0} (z-z_0)^n\right]}
        \end{eqnarray*}
        Let $z_0=0$. For $f(z)=(1+z)^\alpha$, we know
        \begin{eqnarray*}
             \left.\frac{1}{n!}\frac{d^n (1+z)^\alpha}{dz^n}\right|_{z=0}&=&\left.{\alpha \choose n}(1+z)^{\alpha-n}\right|_{z=0} \\
             &=& {\alpha \choose n}
        \end{eqnarray*}
        Thus, the Taylor series for $f(z)=(1+z)^\alpha$ is
        \begin{eqnarray*}
            (1+z)^\alpha&=&1+\sum\limits_{n=1}^{\infty}{\left[{\alpha \choose n}z^n\right]} \\
                        &=&\sum\limits_{n=0}^{\infty}{\left[{\alpha \choose n}z^n\right]}
        \end{eqnarray*}
    \end{proof}

    \begin{eqnarray*}
        f(z) &=& \sum\limits_{n=0}^{\infty}{a_n z^n} \\
        f(-z) &=& \sum\limits_{n=0}^{\infty}{a_n (-z)^n} \\
        \sum\limits_{n=0}^{\infty}{a_n z^n} &=&  \sum\limits_{n=0}^{\infty}{a_n (-z)^n} = \sum\limits_{n=0}^{\infty}{a_n (-1)^{n}z^n} \\
        &\implies& (-1)^{n} = 1 \text{ or } a_n=0 \\
        &\implies& a_n = 0 \text{ if } n \text{ is odd }
    \end{eqnarray*}
    For an odd function, $f(z) = -f(-z)$ for all $z$.
    \begin{eqnarray*}
        f(z) &=& \sum\limits_{n=0}^{\infty}{a_n z^n} \\
        -f(-z) &=& \sum\limits_{n=0}^{\infty}{-a_n (-z)^n} \\
        \sum\limits_{n=0}^{\infty}{a_n z^n} &=&  \sum\limits_{n=0}^{\infty}{-a_n (-z)^n} = \sum\limits_{n=0}^{\infty}{a_n (-1)^{n+1}z^n} \\
        &\implies& (-1)^{n+1} = 1 \text{ or } a_n=0 \\
        &\implies& a_n = 0 \text{ if } n \text{ is even }
